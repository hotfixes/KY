        var currentIdx = 0;
        var imgEl = document.getElementById("main-img");
        var dotsBox = document.getElementById("dots-box");

        function initSlider() {
            if (imgList.length === 0) return;
            imgEl.src = imgList[0];
            for(let i=0; i<imgList.length; i++) {
                let dot = document.createElement("div");
                dot.className = i === 0 ? "dot active" : "dot";
                dotsBox.appendChild(dot);
            }
        }

        function changeSlide(direction) {
            if (imgList.length === 0) return;
            currentIdx += direction;
            if (currentIdx >= imgList.length) currentIdx = 0;
            if (currentIdx < 0) currentIdx = imgList.length - 1;

            imgEl.style.opacity = 0;
            setTimeout(function(){
                imgEl.src = imgList[currentIdx];
                imgEl.style.opacity = 1;
            }, 200);

            var dots = document.querySelectorAll(".dot");
            dots.forEach((d, index) => {
                if(index === currentIdx) d.classList.add("active");
                else d.classList.remove("active");
            });
        }
